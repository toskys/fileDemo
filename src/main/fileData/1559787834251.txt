1.建表
-- 创始担保支付请求表
CREATE TABLE preauthorize_req(  
  id INT(10) NOT NULL AUTO_INCREMENT COMMENT '担保支付单序号',
  merchant_code			          VARCHAR(32) COMMENT '商户号',
  terminal_code			          VARCHAR(32) COMMENT '终端号',
  system_trace_num		          VARCHAR(32) COMMENT '系统跟踪号',
  transaction_amount		          INT(10) COMMENT '交易金额',
  transaction_currency_code	          VARCHAR(32) COMMENT '交易币种',
  merchant_order_id		          VARCHAR(32) COMMENT '商户订单号',
  merchant_remark		          VARCHAR(32) COMMENT '商户备注',
  pay_mode			          VARCHAR(32) COMMENT '支付方式',
  pay_code			          VARCHAR(32) COMMENT '支付码',
  attached_data			          VARCHAR(32) COMMENT '附加信息',
  PRIMARY KEY (id)
);
-- 创始担保支付响应表
CREATE TABLE preauthorize_resp(  
  id INT(10) NOT NULL AUTO_INCREMENT COMMENT '担保支付单序号',
  err_code 			  VARCHAR(16) COMMENT '错误代码',
  err_info			          VARCHAR(32) COMMENT '错误说明',
  transaction_time                        VARCHAR(32) COMMENT '交易时间',
  transaction_date                        VARCHAR(32) COMMENT '交易日期',
  settlement_date                         VARCHAR(32) COMMENT '结算日期',
  retrieval_ref_num                       VARCHAR(32) COMMENT '检索参考号',
  authorization_code                      VARCHAR(32) COMMENT '授权码',
  actual_transaction_amount               DECIMAL(10,2) COMMENT '营销联盟优惠后交易金额',
  amount                                  VARCHAR(32) COMMENT '实际支付金额',
  order_id                                VARCHAR(32) COMMENT '订单号',
  marketing_alliance_discount_instruction VARCHAR(32) COMMENT '营销联盟优惠说明',
  third_party_discount_instruction        VARCHAR(32) COMMENT '第三方优惠说明',
  thirdpartyname                          VARCHAR(32) COMMENT '第三方名称',
  thirdpartybuyerid                       VARCHAR(32) COMMENT '第三方买家Id',
  thirdpartybuyerusername                 VARCHAR(32) COMMENT '第三方买家用户名',
  PRIMARY KEY (id)
);
-- 担保完成请求表
CREATE TABLE complete_authorization_req(  
  id INT(10) NOT NULL AUTO_INCREMENT COMMENT '担保完成单序号',
  merchant_code			          VARCHAR(32) COMMENT '商户号',
  terminal_code			          VARCHAR(32) COMMENT '终端号',
  system_trace_num		          VARCHAR(32) COMMENT '系统跟踪号',
  transaction_amount		          INT(10) COMMENT '交易金额',
  transaction_currency_code	          VARCHAR(32) COMMENT '交易币种',
  authorization_code		          VARCHAR(32) COMMENT '授权码',
  original_transaction_date		          VARCHAR(32) COMMENT '原交易日期',
  original_system_trace_num			          VARCHAR(32) COMMENT '原系统跟踪号',
  PRIMARY KEY (id)
);
-- 担保完成响应表
CREATE TABLE complete_authorization_resp(  
  id INT(10) NOT NULL AUTO_INCREMENT COMMENT '担保支付单序号',
  err_code 			  VARCHAR(16) COMMENT '错误代码',
  err_info			          VARCHAR(32) COMMENT '错误说明',
  transaction_time                        VARCHAR(32) COMMENT '交易时间',
  transaction_date                        VARCHAR(32) COMMENT '交易日期',
  settlement_date                         VARCHAR(32) COMMENT '结算日期',
  retrieval_ref_num                       VARCHAR(32) COMMENT '检索参考号',
  third_party_name                          VARCHAR(32) COMMENT '第三方名称',
  PRIMARY KEY (id)
);
--银行卡支付请求表
CREATE TABLE bank_card_req(  
  id INT(10) NOT NULL AUTO_INCREMENT COMMENT '银行卡支付单序号',
  application_type			          VARCHAR(2) COMMENT '应用类型',
  pos_machine_number			          VARCHAR(8) COMMENT 'POS机号',
  pos_employee_number		          VARCHAR(8) COMMENT 'POS员工号',
  trading_type_mark		          INT(2) COMMENT '交易类型标志',
  amount_of_money	          VARCHAR(12) COMMENT '金额',
  original_transaction_date		          VARCHAR(8) COMMENT '原交易日期',
  original_transaction_reference_number		          VARCHAR(12) COMMENT '原交易参考号',
  original_certificate_number			          VARCHAR(6) COMMENT '原凭证号',
  lrc_check			          VARCHAR(3) COMMENT 'LRC校验',
  serial_code			          VARCHAR(50) COMMENT '串码',
  order_id			          VARCHAR(50) COMMENT '银商订单号',
  erp_order_number			          VARCHAR(50) COMMENT 'ERP订单号',
  dimensional_code_id			          VARCHAR(32) COMMENT '无硬件C扫B二维码ID',
  PRIMARY KEY (id)
);
--银行卡支付响应表
CREATE TABLE bank_card_resp(

  return_code 			  VARCHAR(2) COMMENT '返回码',

  card_number                        VARCHAR(20) COMMENT '卡号',
  voucher_no                        VARCHAR(6) COMMENT '凭证号',
  amount                         VARCHAR(12) COMMENT '实付金额',
  err_info                       VARCHAR(40) COMMENT '错误说明',
  merchant_number                      VARCHAR(15) COMMENT '商户号',
  terminal_code               VARCHAR(8) COMMENT '终端号',
  batch_number                                  VARCHAR(6) COMMENT '批次号',
  transaction_date                                VARCHAR(4) COMMENT '交易日期',
  transaction_time VARCHAR(6) COMMENT '交易时间',
  transaction_reference_number        VARCHAR(12) COMMENT '交易参考号',
  authorization_code                          VARCHAR(6) COMMENT '授权号',
  settlement_date                       VARCHAR(4) COMMENT '清算日期',
  lrc_check                 VARCHAR(3) COMMENT 'LRC校验',
  preferential_amount                 VARCHAR(12) COMMENT '优惠金额',
  card_type                 VARCHAR(2) COMMENT '卡类型',
  third_party_discount_instruction                 VARCHAR(200) COMMENT '第三方优惠说明',
  third_party_platform                 VARCHAR(50) COMMENT '第三方平台',
  order_id                 VARCHAR(50) COMMENT '银商订单号',
  erp_order_number                 VARCHAR(50) COMMENT 'ERP订单号',
  payment_method                 VARCHAR(1) COMMENT '支付方式',
  query_result_status                 VARCHAR(1) COMMENT '查询结果状态',
  query_result_info               VARCHAR(50) COMMENT '查询结果描述',
  two_dimensional_code                 VARCHAR(200) COMMENT '无硬件C扫B二维码',
  billing_time                 VARCHAR(8) COMMENT '无硬件C扫B账单时间',
  order_status                 VARCHAR(20) COMMENT '无硬件C扫B订单状态',
  PRIMARY KEY (id)
);
